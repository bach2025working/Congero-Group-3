#
# Specifies whether to store objects referenced by price objects
# in memory in the CM cache when the CM starts. Objects referenced
# by price objects include BEIDs, GLIDs, tax suppliers, 
# event types, service types, and RUMs. 
#
#
# The value for this entry can be:
#
#    0 = The CM caches objects that are referenced by price objects 
#        in the CM cache every time a price list is committed. 
#         
#        In a production system where you rarely modify your price
#        list, set the flag to 0. This reserves the CM cache
#        for other uses and eliminates the need to restart the
#        CM to update the cache if you change the price list. 
#         
#    1 = (Default) The CM caches objects that are referenced by price 
#         objects one time when the CM starts. 
#
#        In a test environment where you modify your price list
#        often, set the flag to 1. This makes saving the  
#        price list faster since there is no need to load price 
#        reference objects every time you commit the price list
#        to the database. However, you must restart the CM after
#        your price list changes are committed. 
#
# Important: The objects that are referenced by price objects are 
#            created every time the CM starts. If these objects are 
#            changed in the database, the CM must be restarted to 
#            create the new information to place into the cache. 
#
#========================================================================

#
# Specifies whether to store objects referenced by price objects
# in memory in the CM cache when the CM starts. Objects referenced
# by price objects include BEIDs, GLIDs, tax suppliers,
# event types, service types, and RUMs.
#
#
# The value for this entry can be:
#
#    0 = The CM caches objects that are referenced by price objects
#        in the CM cache every time a price list is committed.
#
#        In a production system where you rarely modify your price
#        list, set the flag to 0. This reserves the CM cache
#        for other uses and eliminates the need to restart the
#        CM to update the cache if you change the price list.
#
#    1 = (Default) The CM caches objects that are referenced by price
#         objects one time when the CM starts.
#
#        In a test environment where you modify your price list
#        often, set the flag to 1. This makes saving the
#        price list faster since there is no need to load price
#        reference objects every time you commit the price list
#        to the database. However, you must restart the CM after
#        your price list changes are committed.
#
# Important: The objects that are referenced by price objects are
#            created every time the CM starts. If these objects are
#            changed in the database, the CM must be restarted to
#            create the new information to place into the cache.

   228	*
   229	*******************************************************************/
   230	static int64
   231	convert_to_epoch(
   232	       const char              *date_str)
   233	{
   234	       int month, day, year;
   235
   236	       if (sscanf(date_str, "%d-%d-%d", &month, &day, &year) != 3) {
   237	           return -1;
   238	       }
   239
   240	       if (month < 1 || month > 12 || day < 1 || day > 31 || year < 1900) {
   241	           return -1;
   242	       }
   243
   244	           struct tm tm = {0};
   245	           tm.tm_year = year - 1900;
   246	           tm.tm_mon = month - 1;
   247	           tm.tm_mday = day;
   248	           tm.tm_hour = 0;
   249	           tm.tm_min = 0;
   250	           tm.tm_sec = 0;
   251	           tm.tm_isdst = -1;
   252
   253	           time_t epoch = mktime(&tm);
   254	           if (epoch == -1) {
   255	               return -1;
   256	           }
   257
   258	           return (int64) epoch;
   259	}

   261	/*******************************************************************
   262	* search_login:
   263	*
   264	*******************************************************************/
   265	static void
   266	search_login(
   267	       pcm_context_t           *ctxp,
   268	       int32                   flags,
   269	       poid_t                  *a_pdp,
   270	       char                    *login,
   271	       pin_flist_t             **service_flist,
   272	       poid_t                  **service_poid,
   273	       poid_t                  **account_obj,
   274	       pin_errbuf_t            *ebufp)
   275	{
   276
   277	       // check and clear err buff
   278	       if (PIN_ERR_IS_ERR(ebufp))
   279	           return;
   280	       PIN_ERR_CLEAR_ERR(ebufp);
   281
   282	       // create a search flist
   283	       pin_flist_t *search_flist = PIN_FLIST_CREATE(ebufp);
   284	       pin_flist_t *vp = NULL;
   285
   286	       pin_cookie_t cookie = NULL;
   287	       pin_flist_t * ret_flist = NULL;
   288	       int32 element_id;
   289
   290	       /*
   291	        * Retrieve the database number
   292	        */
   293	       int64 db = PIN_POID_GET_DB(a_pdp);
   294
   295	       /*
   296	        * Create a search object, set flags, template and the arguments
   297	        */
   298	       poid_t * pdp = PIN_POID_CREATE(db, "/search", (int64)-1, ebufp);
   299	       PIN_FLIST_FLD_PUT(search_flist, PIN_FLD_POID, (void *)pdp, ebufp);
   300
   301	       u_int uint_val = SRCH_DISTINCT;
   302	       PIN_FLIST_FLD_SET(search_flist, PIN_FLD_FLAGS, (void *)&uint_val, ebufp);
   303
   304	       PIN_FLIST_FLD_SET(search_flist, PIN_FLD_TEMPLATE,
   305	                         (void *) "select X from /service where F1 = V1 ", ebufp);
   306
   307	       vp = PIN_FLIST_ELEM_ADD(search_flist, PIN_FLD_ARGS, 1, ebufp);
   308	       PIN_FLIST_FLD_SET(vp, PIN_FLD_LOGIN, login, ebufp);
   309
   310	       /*
   311	        * Create PIN_FLD_RESULTS and set a NULL poid in it
   312	        */
   313	       vp = PIN_FLIST_ELEM_ADD(search_flist, PIN_FLD_RESULTS, 0, ebufp);
   314	       PIN_FLIST_FLD_SET(vp, PIN_FLD_POID, (void *)NULL, ebufp);
   315	       PIN_FLIST_FLD_SET(vp, PIN_FLD_ACCOUNT_OBJ, (void *)NULL, ebufp);
   316	       // note, consider adding a pin_fld_account_obj
   317
   318	       PIN_ERR_LOG_FLIST(PIN_ERR_LEVEL_DEBUG,
   319	                         "search_login search flist", search_flist);
   320
   321	       /*
   322	        * Call the PCM_OP_SEARCH
   323	        */
   324	       PCM_OP(ctxp, PCM_OP_SEARCH, 0, search_flist, &ret_flist, ebufp);
   325
   326	       /*
   327	        * Use an internal return flist for the PCM call, when we retrieve the fields
   328	        * from PIN_FLD_RESULTS, use TAKE since we are destroying the return flist
   329	        */
   330	       cookie = NULL;
   331	       vp = PIN_FLIST_ELEM_GET_NEXT(ret_flist, PIN_FLD_RESULTS, &element_id, 1, &cookie, ebufp);
   332
   333	       /*
   334	        * In case PCM_OP_SEARCH fails/returns NULL, exit the function
   335	        */
   336	       if (ret_flist == NULL || vp == NULL) {
   337	           PIN_FLIST_DESTROY_EX(&search_flist, NULL);
   338
   339	           if (ret_flist != NULL) {
   340	               PIN_FLIST_DESTROY_EX(&ret_flist, NULL);
   341	           }
   342
   343	           return;
   344	       }
   345
   346	       *service_poid = PIN_FLIST_FLD_TAKE(vp, PIN_FLD_POID, 0, ebufp);
   347	       *account_obj = PIN_FLIST_FLD_TAKE(vp, PIN_FLD_ACCOUNT_OBJ, 0, ebufp);
   348
   349	       /*
   350	        * Cleaning
   351	        */
   352	       PIN_FLIST_DESTROY_EX(&search_flist, NULL);
   353	       PIN_FLIST_DESTROY_EX(&ret_flist, NULL);
   354
   355	       if (PIN_ERR_IS_ERR(ebufp)) {
   356	           PIN_ERR_LOG_EBUF(PIN_ERR_LEVEL_ERROR,
   357	                           "search_login error", ebufp);
   358	       }
   359
   360	       return;
   361	}

   363	static void
   364	fm_nai_act_rate(
   365	       pcm_context_t           *ctxp,
   366	       int32                   flags,
   367	       poid_t                  *a_pdp,
   368	       poid_t                  *account_obj,
   369	       char                    *date,
   370	       char                    *descr,
   371	       char                    *model_code,
   372	       char                    *trans_id,
   373	       char                    *prompt_txt,
   374	       int64                   *tokens_in,
   375	       int64                   *tokens_out,
   376	       pin_flist_t             **r_flistpp,
   377	       pin_errbuf_t            *ebufp)
   378	{
   379	       /*
   380	        * Some Notes:
   381	        *    use PUT for poids that are from PIN_POID_CREATE
   382	        *    certain functions requires int64, some int32, some needs pointers
   383	        */
   384
   385	       int64 one = 1;
   386	       int64 neg_one = -1;
   387	       int64 zero = 0;
   388
   389	       if (PIN_ERR_IS_ERR(ebufp)) {
   390	               PIN_ERR_LOG_MSG(PIN_ERR_LEVEL_ERROR, "ERROR BUFFER NOT EMPTY IN BAL FUNCTION");
   391	               return;
   392	       }
   393	       PIN_ERR_CLEAR_ERR(ebufp);
   394
   395	       /*
   396	        * If convert_to_epoch fails, it will return -1
   397	        */
   398	       int64 timestamp = convert_to_epoch(date);
   399	       if (timestamp == -1) {
   400	           *r_flistpp = PIN_FLIST_CREATE(ebufp);
   401	           poid_t *err_poid = PIN_POID_CREATE(1, "/error", -1, ebufp);
   402	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_POID, err_poid, ebufp);
   403	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_ERROR_CODE, (void *) "1", ebufp);
   404	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_ERROR_DESCR, (void *) "Invalid information: date (PIN_FLD_NAME)", ebufp);
   405
   406	           return;
   407	       }
   408
   409	       int64 db = PIN_POID_GET_DB(a_pdp);
   410
   411	       pin_flist_t * cust_flist = PIN_FLIST_CREATE(ebufp);
   412	       poid_t * pdp = NULL;
   413
   414	       // 0 PIN_FLD_POID POID [0] 0.0.0.1 /plan ACCOUNT_NUM 0
   415	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_POID, (void *)account_obj, ebufp);
   416
   417	       // 0 PIN_FLD_OBJ_TYPE STR [0] "/usageg3"
   418	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_OBJ_TYPE, (void *) "/usageg3", ebufp);
   419
   420	       // 0 PIN_FLD_PROGRAM_NAME STR [0] "load_session.nap"
   421	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_PROGRAM_NAME, (void *) "load_session.nap", ebufp);
   422
   423	       // 0 PIN_FLD_FLAGS INT [0] 0
   424	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_FLAGS, &zero, ebufp);
   425
   426	       // 0 PIN_FLD_START_T TSTAMP [0] (1756133389)
   427	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_START_T, &timestamp, ebufp);
   428	       // 0 PIN_FLD_END_T TSTAMP [0] (1756133389)
   429	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_END_T, &timestamp, ebufp);
   430
   431	       // 0 PIN_FLD_DESCR STR [0] "MYDESCR"
   432	       PIN_FLIST_FLD_SET(cust_flist, PIN_FLD_DESCR, (void *)descr, ebufp);
   433
   434	       // 0 PIN_FLD_INHERITED_INFO SUBSTRUCT [0]
   435	       pin_flist_t * inherited_info = PIN_FLIST_ELEM_ADD(cust_flist, PIN_FLD_INHERITED_INFO, 0, ebufp);
   436
   437	       // 1     PIN_FLD_INFO_NAI3 SUBSTRUCT [0]
   438	       pin_flist_t * nextaig3 = PIN_FLIST_ELEM_ADD(inherited_info, PIN_FLD_INFO_NAI3, 0, ebufp);
   439
   440	       // 2         PIN_FLD_MODELTYPE_NAI3 STR [0] "3.0 or 3.5"
   441	       PIN_FLIST_FLD_SET(nextaig3, PIN_FLD_MODELTYPE_NAI3, (void *)model_code, ebufp);
   442
   443	       // 2         PIN_FLD_TRANSACTION_ID STR [0] "MYTRANSID"
   444	       PIN_FLIST_FLD_SET(nextaig3, PIN_FLD_TRANSACTION_ID, (void *)trans_id, ebufp);
   445
   446	       // 2         PIN_FLD_PROMPT_NAI3 STR [0] "MYPROMPT"
   447	       PIN_FLIST_FLD_SET(nextaig3, PIN_FLD_PROMPT_NAI3, (void *)prompt_txt, ebufp);
   448
   449	       // 2         PIN_FLD_INPUT_TOKENS_NAI3 INT [0] 205000
   450	       PIN_FLIST_FLD_SET(nextaig3, PIN_FLD_INPUT_TOKENS_NAI3, (void *)tokens_in, ebufp);
   451
   452	       // 2         PIN_FLD_OUTPUT_TOKENS_NAI3 INT [0] 200
   453	       PIN_FLIST_FLD_SET(nextaig3, PIN_FLD_OUTPUT_TOKENS_NAI3, (void *)tokens_out, ebufp);
   454
   455	       pin_flist_t * pcm_return_flist = PIN_FLIST_CREATE(ebufp);
   456	       PCM_OP(ctxp, PCM_OP_ACT_LOAD_SESSION, flags, cust_flist, &pcm_return_flist, ebufp); //dont use r_flistpp
   457
   458	       *r_flistpp = PIN_FLIST_CREATE(ebufp);
   459
   460	       /*
   461	        * If theres an error, sends error code 1 with the message
   462	        * else, sends error code 0 and the poid
   463	        */
   464	       if (PIN_ERR_IS_ERR(ebufp)) {
   465	           PIN_ERR_LOG_EBUF(PIN_ERR_LEVEL_ERROR,
   466	                       "op_nai_read error", ebufp);
   467	           PIN_ERR_LOG_EBUF(PIN_ERR_LEVEL_ERROR, "fm_nai_act_rate PCM_OP error", ebufp);
   468
   469	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_ERROR_CODE, (void *)"1", ebufp);
   470	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_ERROR_DESCR, (void *)"Issue with PCM_OP_ACT_LOAD_SESSION", ebufp);
   471	       } else {
   472	           PIN_ERR_LOG_FLIST(PIN_ERR_LEVEL_DEBUG, "PCM_OP_ACT_LOAD_SESSION output", pcm_return_flist);
   473
   474	           poid_t *cust_poid = PIN_FLIST_FLD_GET(pcm_return_flist, PIN_FLD_POID, 0, ebufp);
   475	               if (cust_poid != NULL) {
   476	                   PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_POID, (void *)cust_poid, ebufp);
   477	           }
   478
   479	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_ERROR_CODE, (void *) "0", ebufp);
   480	           PIN_FLIST_FLD_SET(*r_flistpp, PIN_FLD_ERROR_DESCR, (void *) "Act rated successfully", ebufp);
   481
   482	       }
   483
   484	       PIN_FLIST_DESTROY_EX(&cust_flist, NULL);
   485	       PIN_FLIST_DESTROY_EX(&pcm_return_flist, NULL);
   486	       return;
   487	}
